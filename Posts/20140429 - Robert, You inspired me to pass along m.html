Robert, You inspired me to pass along my approach to the problem so far, it might give you some ideas. But I still think this is going to get solved by the freebase devs so that the API gives useable results. Note also this thread: https://groups.google.com/d/msg/freebase-discuss/h_12vauCqCY/nXKYATIi7WgJ

import urllib
import urllib2
import sys
from rdflib import Graph

api_key =  "YourKeyHere" 
freebase_url = 'https://www.googleapis.com/freebase/v1/rdf'
topic_id = '/m/046ym'
params = {'key': api_key}
url = freebase_url + topic_id + '?' + urllib.urlencode(params)
query_result = urllib2.urlopen(url)
rdf = ''

def collapse_quotations(str):
    lst = str.split('"')
    lst.insert(1, '"')
    lst.insert(-1, '"')
    return ''.join(lst)

for line in query_result:
    if line.count('"') > 2:
        line = collapse_quotations(line)
    rdf += line.replace('\\x','\\\\x').replace('$','')

g = Graph()
g.parse(data=rdf, format='n3')

for s, p, o in g: 
    print s, p, o
